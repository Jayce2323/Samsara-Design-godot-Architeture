# How modifiers apply to items
func apply_modifier_effects(item: Item, modifier: int, category: ModifierCategory) -> void:
    var effects:  Dictionary = get_effects_for_modifier(modifier, category)
    
    for stat_key in effects.keys():
        var value = effects[stat_key]
        
        match stat_key:
            # Multiplicative effects
            "all_stats_mult", "durability_mult", "value_mult": 
                item.set_stat_multiplier(stat_key, value)
            
            # Additive percentage effects
            "durability_mod", "armor_value_mod", "weapon_value_mod": 
                item.add_stat_modifier(stat_key, value)
            
            # Flat additive effects
            "appearance_mod", "prestige_bonus", "fame_on_display":
                item.add_flat_modifier(stat_key, value)
            
            # Boolean flags
            "functional", "repairable", "salvageable": 
                item.set_flag(stat_key, value)
            
            # Special handlers
            "contact_damage_heat", "contact_damage_cold", "contact_damage_lightning":
                item. set_contact_damage(stat_key. replace("contact_damage_", ""), value)

func get_connotation(modifier: int, category: ModifierCategory) -> ModifierConnotation: 
    match category:
        ModifierCategory.PHYSICAL_CONDITION: 
            return PHYSICAL_CONDITION_CONNOTATION[modifier]
        ModifierCategory.QUALITY_TIER:
            return QUALITY_TIER_CONNOTATION[modifier]
        # ... etc for each category
    return ModifierConnotation.NEUTRAL

# From Social Sim6.txt:  Disposition-components:  First Impressions:  Appearance
# Items-worn - Display Stats link to Item adjectives -> Affects "Appearance" +,0,-(Ratings are general +,- = 1)

func calculate_appearance_from_modifiers(equipped_items: Array) -> int:
    var total_appearance:  int = 0
    
    for item in equipped_items: 
        for modifier in item.modifiers:
            var effects = get_effects_for_modifier(modifier. type, modifier.category)
            if effects.has("appearance_mod"):
                total_appearance += effects["appearance_mod"]
    
    return total_appearance

# From UEG TYPE2.txt: Connotation Algebra
# (+) + (+) = (+), (-) + (-) = (+), (+) + (-) = (+)

func calculate_item_connotation(item: Item) -> ModifierConnotation: 
    var positive_count:  int = 0
    var negative_count: int = 0
    
    for modifier in item.modifiers:
        var connotation = get_connotation(modifier.type, modifier. category)
        match connotation: 
            ModifierConnotation.POSITIVE: 
                positive_count += 1
            ModifierConnotation.NEGATIVE:
                negative_count += 1
    
    # Apply connotation algebra
    if positive_count > 0 and negative_count == 0:
        return ModifierConnotation.POSITIVE
    elif negative_count > 0 and positive_count == 0:
        return ModifierConnotation.NEGATIVE
    elif positive_count > 0 and negative_count > 0:
        return ModifierConnotation. POSITIVE  # (+) + (-) = (+)
    else:
        return ModifierConnotation.NEUTRAL

FRESH → (time) → RECENT → (time) → AGED → (time) → STALE → (time) → SPOILED → (time) → ROTTEN → (time) → PUTRID

FRESH → (salt/pickle) → PRESERVED (halts decay)
FRESH → (intentional ferment) → FERMENTED (different decay path)
FRESH → (dry) → MUMMIFIED (halts decay)
FRESH → (freeze) → FROZEN (halts decay until thaw)

    all_modifiers.merge(PHYSICAL_NEUTRAL)

    return all_modifiers

    var all_modifiers: Dictionary = {}
    all_modifiers. merge(SENSORY_VISUAL_POSITIVE)
    all_modifiers.merge(SENSORY_VISUAL_NEUTRAL)
    all_modifiers.merge(SENSORY_VISUAL_NEGATIVE)
    all_modifiers.merge(SENSORY_OLFACTORY)
    all_modifiers.merge(SENSORY_TACTILE)
    all_modifiers.merge(SENSORY_AUDITORY)
    return all_modifiers

    if not _modifiers_by_category.has(category):
        _modifiers_by_category[category] = []
    
    for key in modifiers.keys():
        var mod_data = modifiers[key]
        mod_data["_key"] = key
        mod_data["_category"] = category
        _all_modifiers[key] = mod_data
        _modifiers_by_category[category].append(key)
        
        if mod_data.has("connotation"):
            _modifiers_by_connotation[mod_data["connotation"]].append(key)
        
        if mod_data.has("domain"):
            var domain = mod_data["domain"]
            if not _modifiers_by_domain. has(domain):
                _modifiers_by_domain[domain] = []
            _modifiers_by_domain[domain]. append(key)

func calculate_item_connotation(item_modifiers:  Array) -> AdjEffectEnums. AdjConnotation: 
    var positive_count: int = 0
    var negative_count: int = 0
    
    for mod_key in item_modifiers:
        var mod_data = get_modifier(mod_key)
        if mod_data.has("connotation"):
            match mod_data["connotation"]: 
                AdjEffectEnums.AdjConnotation. POSITIVE:
                    positive_count += 1
                AdjEffectEnums.AdjConnotation.NEGATIVE: 
                    negative_count += 1
    
    # UEG Connotation Algebra (from UEG TYPE2. txt):
    # (+) + (+) = (+)
    # (-) + (-) = (+)  [double negative = positive]
    # (+) + (-) = (+)  [positive dominates]
    # (0) + (x) = (x)  [neutral is identity]
    
    if positive_count > 0 and negative_count == 0:
        return AdjEffectEnums.AdjConnotation. POSITIVE
    elif negative_count > 0 and positive_count == 0:
        # Single negative remains negative
        if negative_count == 1:
            return AdjEffectEnums.AdjConnotation.NEGATIVE
        # Double negative becomes positive
        else:
            return AdjEffectEnums.AdjConnotation.POSITIVE
    elif positive_count > 0 and negative_count > 0:
        # Positive dominates mixed
        return AdjEffectEnums. AdjConnotation.POSITIVE
    else:
        return AdjEffectEnums.AdjConnotation.NEUTRAL

func calculate_social_appearance_modifier(item_modifiers: Array) -> int:
    """Calculate total appearance modifier for Social Sim integration."""
    var total:  int = 0
    
    for mod_key in item_modifiers:
        var mod_data = get_modifier(mod_key)
        if mod_data. has("effects"):
            for effect in mod_data["effects"]:
                if effect["target"] == AdjEffectEnums.AdjEffectTarget.APPEARANCE_GENERAL:
                    if effect["operation"] == AdjEffectEnums.AdjEffectOperation. ADD_FLAT: 
                        total += effect["value"]
                    elif effect["operation"] == AdjEffectEnums. AdjEffectOperation.SUBTRACT_FLAT:
                        total -= effect["value"]
    
    return total

func apply_modifier_to_item(item: Object, modifier_key: String) -> bool:
    var mod_data = get_modifier(modifier_key)
    if mod_data.is_empty():
        push_warning("Modifier not found: " + modifier_key)
        return false
    
    # Check application context
    if mod_data.has("application"):
        var app = mod_data["application"]
        if app == AdjEffectEnums.AdjApplication.ACTOR_ONLY:
            push_warning("Modifier " + modifier_key + " is actor-only, cannot apply to item")
            return false
    
    # Check material requirements
    if mod_data.has("material_requirement"):
        var item_material = _get_item_material(item)
        if item_material and item_material not in mod_data["material_requirement"]: 
            push_warning("Item material " + item_material + " not valid for modifier " + modifier_key)
            return false
    
    # Apply effects
    if mod_data.has("effects"):
        for effect in mod_data["effects"]: 
            _apply_effect(item, effect)
    
    # Apply special effects
    if mod_data.has("special_effects"):
        _apply_special_effects(item, mod_data["special_effects"])
    
    # Track modifier on item
    _add_modifier_to_item(item, modifier_key)
    
    return true

func remove_modifier_from_item(item: Object, modifier_key:  String) -> bool:
    var mod_data = get_modifier(modifier_key)
    if mod_data.is_empty():
        return false
    
    # Check if removable
    if mod_data.has("removal_method"):
        if mod_data["removal_method"] == AdjEffectEnums.AdjRemovalMethod.NONE:
            push_warning("Modifier " + modifier_key + " cannot be removed")
            return false
    
    # Reverse effects
    if mod_data.has("effects"):
        for effect in mod_data["effects"]: 
            _reverse_effect(item, effect)
    
    # Remove from item tracking
    _remove_modifier_from_item(item, modifier_key)
    
    return true

func _apply_effect(item: Object, effect:  Dictionary) -> void:
    var target = effect["target"]
    var operation = effect["operation"]
    var value = effect["value"]
    
    var current = _get_item_stat(item, target)
    var new_value = _calculate_new_value(current, operation, value)
    _set_item_stat(item, target, new_value)

func _reverse_effect(item: Object, effect: Dictionary) -> void:
    var target = effect["target"]
    var operation = effect["operation"]
    var value = effect["value"]
    
    var current = _get_item_stat(item, target)
    var new_value = current
    
    # Reverse the operation
    match operation:
        AdjEffectEnums.AdjEffectOperation.ADD_FLAT:
            new_value = current - value
        AdjEffectEnums. AdjEffectOperation.SUBTRACT_FLAT:
            new_value = current + value
        AdjEffectEnums. AdjEffectOperation.ADD_PERCENT:
            new_value = current / (1.0 + value / 100.0)
        AdjEffectEnums. AdjEffectOperation.SUBTRACT_PERCENT:
            new_value = current / (1.0 - value / 100.0)
        AdjEffectEnums.AdjEffectOperation.MULTIPLY:
            if value != 0:
                new_value = current / value
        # SET_VALUE and SET_FLAG cannot be reversed without original value
    
    _set_item_stat(item, target, new_value)

func _calculate_new_value(current, operation: AdjEffectEnums.AdjEffectOperation, value):
    match operation: 
        AdjEffectEnums.AdjEffectOperation. ADD_FLAT: 
            return current + value
        AdjEffectEnums. AdjEffectOperation.SUBTRACT_FLAT:
            return current - value
        AdjEffectEnums.AdjEffectOperation.ADD_PERCENT:
            return current * (1.0 + value / 100.0)
        AdjEffectEnums. AdjEffectOperation.SUBTRACT_PERCENT:
            return current * (1.0 - value / 100.0)
        AdjEffectEnums.AdjEffectOperation. MULTIPLY:
            return current * value
        AdjEffectEnums.AdjEffectOperation.SET_VALUE:
            return value
        AdjEffectEnums.AdjEffectOperation.SET_FLAG:
            return value
        AdjEffectEnums.AdjEffectOperation.CLEAR_FLAG:
            return false
    return current

const MUTUALLY_EXCLUSIVE_MODIFIERS:  Array = [
    # Temperature states - only one at a time
    ["frozen", "cold", "warm", "hot", "burning", "molten"],
    # Moisture states - only one at a time
    ["dry", "damp", "wet", "soaked", "waterlogged"],
    # Contamination levels - only highest applies
    ["clean", "dusty", "dirty", "filthy"],
    # Freshness states - only one at a time
    ["fresh", "recent", "aged", "stale", "spoiled", "rotten", "putrid"],
    # Condition extremes
    ["pristine", "broken", "shattered", "ruined"],
    # Quality tiers - only one at a time
    ["shoddy", "crude", "flimsy", "defective", "common", "serviceable", 
     "standard", "well_made", "fine", "superior", "exceptional", "masterwork",
     "legendary", "divine", "primordial"],
    # Age states
    ["new", "current", "dated", "antique", "ancient", "primeval", "timeless"],
]

func can_add_modifier(item: Object, modifier_key: String) -> bool:
    """Check if modifier can be added to item (compatibility check)."""
    var current_modifiers:  Array = _get_item_modifiers(item)
    
    # Check mutual exclusivity
    for exclusion_group in MUTUALLY_EXCLUSIVE_MODIFIERS:
        if modifier_key in exclusion_group: 
            for existing_mod in current_modifiers:
                if existing_mod in exclusion_group and existing_mod != modifier_key:
                    return false
    
    # Check if already has this modifier
    if modifier_key in current_modifiers:
        return false
    
    return true

func get_conflicting_modifiers(item: Object, modifier_key: String) -> Array:
    """Get list of modifiers that would conflict with adding the new one."""
    var conflicts: Array = []
    var current_modifiers:  Array = _get_item_modifiers(item)
    
    for exclusion_group in MUTUALLY_EXCLUSIVE_MODIFIERS: 
        if modifier_key in exclusion_group:
            for existing_mod in current_modifiers: 
                if existing_mod in exclusion_group and existing_mod != modifier_key: 
                    conflicts.append(existing_mod)
    
    return conflicts

func replace_modifier(item:  Object, old_key: String, new_key: String) -> bool:
    """Remove old modifier and add new one (for upgrades/downgrades)."""
    if remove_modifier_from_item(item, old_key):
        return apply_modifier_to_item(item, new_key)
    return false

func _get_item_modifiers(item: Object) -> Array:
    if item.has_method("get_modifiers"):
        return item.get_modifiers()
    elif "modifiers" in item and item.modifiers is Array:
        return item.modifiers
    return []

func process_modifier_decay(item: Object, delta_time: float) -> Array:
    """Process decay for all modifiers on item.  Returns removed modifiers."""
    var removed:  Array = []
    var current_modifiers = _get_item_modifiers(item).duplicate()
    
    for mod_key in current_modifiers:
        var mod_data = get_modifier(mod_key)
        if mod_data. has("decay_rate") and mod_data["decay_rate"] > 0:
            var decay_amount = mod_data["decay_rate"] * delta_time
            
            # Track decay progress on item
            var decay_key = "_decay_" + mod_key
            var current_decay = 0.0
            if item.has_method("get_meta"):
                current_decay = item.get_meta(decay_key) if item.has_meta(decay_key) else 0.0
            
            current_decay += decay_amount
            
            if current_decay >= 1.0:
                # Modifier has fully decayed
                remove_modifier_from_item(item, mod_key)
                removed.append(mod_key)
                if item.has_method("remove_meta"):
                    item.remove_meta(decay_key)
            else:
                if item.has_method("set_meta"):
                    item.set_meta(decay_key, current_decay)
    
    return removed

func get_modifier_decay_progress(item: Object, modifier_key: String) -> float:
    """Get decay progress (0.0 to 1.0) for a modifier on item."""
    var decay_key = "_decay_" + modifier_key
    if item.has_method("get_meta") and item.has_meta(decay_key):
        return item.get_meta(decay_key)
    return 0.0

func get_total_value_modifier(item_modifiers: Array) -> float:
    """Calculate total multiplicative value modifier from all modifiers."""
    var total_mult:  float = 1.0
    var total_add_percent: float = 0.0
    
    for mod_key in item_modifiers:
        var mod_data = get_modifier(mod_key)
        if mod_data.has("effects"):
            for effect in mod_data["effects"]:
                if effect["target"] == AdjEffectEnums.AdjEffectTarget. ITEM_VALUE:
                    match effect["operation"]: 
                        AdjEffectEnums. AdjEffectOperation. MULTIPLY:
                            total_mult *= effect["value"]
                        AdjEffectEnums.AdjEffectOperation.ADD_PERCENT: 
                            total_add_percent += effect["value"]
                        AdjEffectEnums.AdjEffectOperation. SUBTRACT_PERCENT: 
                            total_add_percent -= effect["value"]
    
    return total_mult * (1.0 + total_add_percent / 100.0)

func get_modifier_summary(modifier_key: String) -> String:
    """Get human-readable summary of modifier effects."""
    var mod_data = get_modifier(modifier_key)
    if mod_data.is_empty():
        return "Unknown modifier"
    
    var summary: String = modifier_key. capitalize() + ": "
    
    if mod_data.has("description"):
        summary += mod_data["description"]
    
    if mod_data. has("effects") and mod_data["effects"]. size() > 0:
        summary += " ["
        var effect_strings: Array = []
        for effect in mod_data["effects"]:
            var target_name = _target_to_stat_name(effect["target"])
            var op_str = ""
            match effect["operation"]: 
                AdjEffectEnums.AdjEffectOperation. ADD_FLAT: 
                    op_str = "+" + str(effect["value"])
                AdjEffectEnums.AdjEffectOperation. SUBTRACT_FLAT:
                    op_str = "-" + str(effect["value"])
                AdjEffectEnums.AdjEffectOperation.ADD_PERCENT:
                    op_str = "+" + str(effect["value"]) + "%"
                AdjEffectEnums.AdjEffectOperation.SUBTRACT_PERCENT:
                    op_str = "-" + str(effect["value"]) + "%"
                AdjEffectEnums.AdjEffectOperation.MULTIPLY:
                    op_str = "×" + str(effect["value"])
            effect_strings.append(target_name + " " + op_str)
        summary += ", ".join(effect_strings) + "]"
    
    return summary

func list_all_modifiers() -> Array:
    """Get list of all registered modifier keys."""
    return _all_modifiers.keys()

func get_modifier_count() -> int:
    """Get total number of registered modifiers."""
    return _all_modifiers.size()

func print_registry_stats() -> void:
    """Print statistics about the modifier registry."""
    print("=== Modifier System Registry Stats ===")
    print("Total modifiers:  ", get_modifier_count())
    print("By category:")
    for cat in _modifiers_by_category.keys():
        print("  ", cat, ": ", _modifiers_by_category[cat].size())
    print("By connotation:")
    print("  Positive: ", get_all_positive_modifiers().size())
    print("  Neutral: ", get_all_neutral_modifiers().size())
    print("  Negative: ", get_all_negative_modifiers().size())

static func get_connotation(modifier_key: String) -> AdjEffectEnums.AdjConnotation: 
    """Get connotation for any modifier key."""
    if PHYSICAL_CONDITION. has(modifier_key):
        return PHYSICAL_CONDITION[modifier_key]
    if QUALITY_TIER.has(modifier_key):
        return QUALITY_TIER[modifier_key]
    if MATERIAL_STATE.has(modifier_key):
        return MATERIAL_STATE[modifier_key]
    if MAGICAL_STATE.has(modifier_key):
        return MAGICAL_STATE[modifier_key]
    if SOCIAL_STATE.has(modifier_key):
        return SOCIAL_STATE[modifier_key]
    if TEMPORAL_STATE.has(modifier_key):
        return TEMPORAL_STATE[modifier_key]
    if SENSORY_STATE.has(modifier_key):
        return SENSORY_STATE[modifier_key]
    
    return AdjEffectEnums.AdjConnotation. NEUTRAL

static func get_all_by_connotation(connotation: AdjEffectEnums.AdjConnotation) -> Array:
    """Get all modifiers with specified connotation."""
    var result: Array = []
    var all_maps = [PHYSICAL_CONDITION, QUALITY_TIER, MATERIAL_STATE, 
                    MAGICAL_STATE, SOCIAL_STATE, TEMPORAL_STATE, SENSORY_STATE]
    
    for map in all_maps:
        for key in map. keys():
            if map[key] == connotation:
                result.append(key)
    
    return result

static func is_positive(modifier_key:  String) -> bool:
    return get_connotation(modifier_key) == AdjEffectEnums.AdjConnotation. POSITIVE

static func is_negative(modifier_key:  String) -> bool:
    return get_connotation(modifier_key) == AdjEffectEnums.AdjConnotation.NEGATIVE

static func is_neutral(modifier_key:  String) -> bool:
    return get_connotation(modifier_key) == AdjEffectEnums.AdjConnotation.NEUTRAL
